437     /**
438      * The Luhn algorithm.
439      *
440      * @name Luhn
441      * @class Luhn
442      * @constructor
443      * @param {String} str Code value
444      */
445      var Luhn = function(str){
446          this.luhnValue = ''
447          this.luhnLength = 0
448 
449          if(typeof str == 'string' && str.length){
450              this.luhnValue = str
451              this.luhnLength = str.length
452          }
453          else {
454              throw 'Code must be a non null string.'
455          }
456 
457          this.toString = function(){
458              return this.luhnValue
459          }
460 
461          /**
462           * Compute modulo 10 using the Luhn algorithm.
463           *
464           * @return {Number} The modulo 10 result
465           */
466          this.modulo10 = function(){
467              var code = this.luhnValue.split('').reverse()
468              var sum = 0
469 
470              for(i in code){
471                  var digit = parseInt(code[i])
472 
473                  if((parseInt(i) + 1) % 2 == 0){
474                      digit = digit * 2
475 
476                      if(digit > 9){
477                          digit = digit - 9
478                      }
479                  }
480 
481                  sum += digit 
482              }
483 
484              return sum % 10
485          }
486 
487          /**
488           * Check if the code if valid or not.
489           *
490           * @return {Number} True if code is valid
491           */
492          this.check = function(){
493              return !this.modulo10()
494          }
495      }

507      var Rib = function(str){
508          this.value = ''
509          this.length = 0

512              this.value = new String(str)
513              this.length = str.length

523          /**
524           * Check if the code if valid or not.
525           *
526           * @return {Number} True if code is valid
527           */
528          this.check = function(){
529              var strAccount = ''
530              var str = this.getNumeroDeCompte()
531              var strRib = ''
532 
533              for(var i in str){
534                  var c = str[i]
535 
536                  if(c.match(/[^0-9]/)){
537                      var intCProv = c.charCodeAt(0) - 64
538 
539                      var intC = ((intCProv + Math.pow(2, (intCProv - 10) / 9 )) % 10)
540                      intC += intCProv > 18 && intCProv < 25 ? 1 : 0
541 
542                      c = new String(intC)
543                  }
544 
545                  strAccount += c
546              }
547 
548              strRib += this.getCodeBanque() + ''
549              strRib += this.getCodeGuichet() + ''
550              strRib += strAccount + ''
551              strRib += this.getCleRib() + ''
552 
553              return modulo(strRib, 97) == 0 && this.length == 23
554          }
555

557              return this.value.substr(0 , 5);
558          }

609          /**
610           * Change the code to follow some rules used for Corsica. In Corsica,
611           * two letters are used into the code: A and B. This letter must be
612           * replaced by a zero and in case of A, substract by 1.000.000 or
613           * 2.000.000 for B.
614           *
615           * @param {String} str The code to adapt
616           * @return {String} The adapted code
617           */
618          this.adapt = function(str){
619              var str = str.substr(0, 13)
620              var substract = 0
621              if(str.match(/[aA]g/)) substract = 1000000
622              if(str.match(/[bB]g/)) substract = 2000000
623              var int = parseInt(str.replace(/ABab/g, '0'))
624 
625              if(substract > 0){
626                  return new String(int - substract)
627              }
628              else {
629                  return str
630              }
631          }

663      /**
664       * Modulo for big value.
665       *
666       * Avoiding some issues with javascript and modulo calculus of big value.
667       *
668       * See http://stackoverflow.com/questions/929910/modulo-in-javascript-large-number
669       *
670       * @lends _private_methods
671       * @param {String} divident
672       * @param {Number} divisor
673       */
674      var modulo = function(divident, divisor) {
675          var cDivident = ''
676          var cRest = ''
677 
678          divident = divident + ''
679 
680          for (var i in divident ) {
681              var cChar = divident[i]
682              var cOperator = cRest + '' + cDivident + '' + cChar
683 
684              if ( cOperator < parseInt(divisor) ) {
685                  cDivident += '' + cChar
686              }
687              else {
688                  cRest = cOperator % divisor
689                  if ( cRest == 0 ) {
690                      cRest = ''
691                  }
692                  cDivident = ''
693              }
694 
695          }
696          cRest += '' + cDivident
697          
698          if (cRest == '') {
699              cRest = 0
700          }
701 
702          return cRest
703      }

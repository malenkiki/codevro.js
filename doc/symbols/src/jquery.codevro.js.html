293         if ( methods[methodOrOptions] ) {
294             return methods[ methodOrOptions ].apply( this, Array.prototype.slice.call( arguments, 1 ));
295         }
296         else if ( typeof methodOrOptions === 'object' || ! methodOrOptions ) {
297             // Default to "init"
298             return methods.init.apply( this, arguments );
299         }
300         else {
301             $.error( 'Method ' +  method + ' does not exist on jQuery.codevro' );
302         }

321     /**
322      * The Luhn algorithm.
323      *
324      * @name Luhn
325      * @class Luhn
326      * @constructor
327      * @param {String} str Code value
328      */
329      var Luhn = function(str){
330          this.luhnValue = ''
331          this.luhnLength = 0
332 
333          if(typeof str == 'string' && str.length){
334              this.luhnValue = str
335              this.luhnLength = str.length
336          }
337          else {
338              throw 'Code must be a non null string.'
339          }
340 
341          this.toString = function(){
342              return this.luhnValue
343          }
344 
345          /**
346           * Compute modulo 10 using the Luhn algorithm.
347           *
348           * @return {Number} The modulo 10 result
349           */
350          this.modulo10 = function(){
351              var code = this.luhnValue.split('').reverse()
352              var sum = 0
353 
354              for(i in code){
355                  var digit = parseInt(code[i])
356 
357                  if((parseInt(i) + 1) % 2 == 0){
358                      digit = digit * 2
359 
360                      if(digit > 9){
361                          digit = digit - 9
362                      }
363                  }
364 
365                  sum += digit 
366              }
367 
368              return sum % 10
369          }
370 
371          /**
372           * Check if the code if valid or not.
373           *
374           * @return {Number} True if code is valid
375           */
376          this.check = function(){
377              return !this.modulo10()
378          }
379      }

407          /**
408           * Check if the code if valid or not.
409           *
410           * @return {Number} True if code is valid
411           */
412          this.check = function(){
413              var strAccount = ''
414              var str = this.getNumeroDeCompte()
415              var strRib = ''
416 
417              for(var i in str){
418                  var c = str[i]
419 
420                  if(c.match(/[^0-9]/)){
421                      var intCProv = c.charCodeAt(0) - 64
422 
423                      var intC = ((intCProv + Math.pow(2, (intCProv - 10) / 9 )) % 10)
424                      intC += intCProv > 18 && intCProv < 25 ? 1 : 0
425 
426                      c = new String(intC)
427                  }
428 
429                  strAccount += c
430              }
431 
432              strRib += this.getCodeBanque() + ''
433              strRib += this.getCodeGuichet() + ''
434              strRib += strAccount + ''
435              strRib += this.getCleRib() + ''
436 
437              return modulo(strRib, 97) == 0 && this.length == 23
438          }

460      /**
461       * Modulo for big value.
462       *
463       * Avoiding some issues with javascript and modulo calculus of big value.
464       *
465       * See http://stackoverflow.com/questions/929910/modulo-in-javascript-large-number
466       *
467       * @lends _private_methods
468       * @param {String} divident
469       * @param {Number} divisor
470       */
471      var modulo = function(divident, divisor) {
472          var cDivident = ''
473          var cRest = ''
474 
475          divident = divident + ''
476 
477          for (var i in divident ) {
478              var cChar = divident[i]
479              var cOperator = cRest + '' + cDivident + '' + cChar
480 
481              if ( cOperator < parseInt(divisor) ) {
482                  cDivident += '' + cChar
483              }
484              else {
485                  cRest = cOperator % divisor
486                  if ( cRest == 0 ) {
487                      cRest = ''
488                  }
489                  cDivident = ''
490              }
491 
492          }
493          cRest += '' + cDivident
494          
495          if (cRest == '') {
496              cRest = 0
497          }
498 
499          return cRest
500      }
